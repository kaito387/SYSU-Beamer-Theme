\section{排版示例}

\begin{frame}{公式与排版}
    线性回归常见目标函数：
    $$
    \min_{\mathbf{w}}\;\frac{1}{n}\sum_{i=1}^{n}\bigl(y_i-\mathbf{w}^{\top}\mathbf{x}_i\bigr)^2 + \lambda\lVert\mathbf{w}\rVert_2^2
    $$
    其中 $\lambda>0$ 控制正则强度。
\end{frame}

\begin{frame}[fragile]{代码示例}
\begin{lstlisting}[language=Python,style=sysucode,emph={solve,lam,n_features}]
def solve(xs, ys, lam=1e-3):
    # ridge closed-form: w = (X^T X + lam I)^(-1) X^T y
    import numpy as np
    X = np.asarray(xs, dtype=float)
    y = np.asarray(ys, dtype=float)
    n_features = X.shape[1]
    A = X.T @ X + lam * np.eye(n_features)
    b = X.T @ y
    return np.linalg.solve(A, b)
\end{lstlisting}
\end{frame}

\section{Listings 进阶示例}

\begin{frame}{\texttt{lstset} 里这些选项如何生效}
    \begin{itemize}
        \item \texttt{keywordstyle}: 作用在语言关键字（如 \texttt{def}, \texttt{return}, \texttt{if}）。
        \item \texttt{stringstyle}: 作用在字符串字面量（如 \texttt{"SYSU"}）。
        \item \texttt{emphstyle}: 只对 \texttt{emph={...}} 中声明的标识符生效。
        \item 局部覆盖：每个 \texttt{lstlisting} 可单独传参数，临时改颜色、行号和 frame。
    \end{itemize}
\end{frame}

\begin{frame}[fragile]{关键字 + 字符串 + emph 对照}
\begin{lstlisting}[language=Python,style=sysucode,emph={train_epoch,loss,batch},emphstyle=\bfseries\color{deepred}]
def train_epoch(model, loader):
    total = 0.0
    for batch in loader:
        pred = model(batch["x"])
        loss = ((pred - batch["y"]) ** 2).mean()
        total += loss.item()
    return "epoch done", total
\end{lstlisting}
\vspace{0.3em}
\small 观察：\texttt{def/for/return} 用 \texttt{keywordstyle}，字符串用 \texttt{stringstyle}，\texttt{train\_epoch/loss/batch} 用 \texttt{emphstyle}。
\end{frame}

\begin{frame}[fragile]{局部覆盖示例（不改全局）}
\begin{lstlisting}[
    language=Python,
    style=focusline,
    frame=lines,
    numbers=right,
    keywordstyle=\color{blue!80!black}\bfseries,
    stringstyle=\color{green!40!black},
    emph={normalize,eps},
    emphstyle=\color{red!75!black}\bfseries
]
def normalize(x, eps=1e-6):
    msg = "safe normalize"
    s = sum(v * v for v in x) ** 0.5
    return [v / (s + eps) for v in x], msg
\end{lstlisting}
\end{frame}

\section{如何继续定制}

\begin{frame}{你可以改哪些地方}
    \begin{enumerate}
        \item 在 \texttt{sysu.sty} 中改主色（\texttt{sysuGreen}）与页脚模板。
        \item 在导言区加学校 logo（或校训）命令，统一 title page 风格。
        \item 针对课程场景新增环境：定理、作业、实验结论等。
        \item 固定 16:9 或 4:3 比例，形成你们组内统一模板。
    \end{enumerate}
\end{frame}
